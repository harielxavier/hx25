rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow anyone to read posts
    match /posts/{postId} {
      allow read: if true;  // Anyone can read posts
      allow write: if request.auth != null && request.auth.token.email_verified;  // Only authenticated users can write
    }
    
    // Your existing rules for other collections
    match /{document=**} {
      allow read, write: if request.auth != null && request.auth.token.email_verified;
    }
  }
}
